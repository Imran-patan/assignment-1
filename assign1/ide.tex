\section{karnaugh-map}
        \begin{karnaugh-map}[4][4][1][$s3s2$][$s1s0$]
        \minterms{1,5,6,7,11,12,13,15}
        \maxterms{0,2,3,4,8,9,10,14}
        \implicant{1}{5}
        \implicant{7}{6}
        \implicant{12}{13}
        \implicant{15}{11}
        \end{karnaugh-map}
